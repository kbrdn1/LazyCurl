use ratatui::{
    prelude::*,
    widgets::{Block, Borders, Paragraph},
};

pub struct MotionBlurSplash {
    frame: usize,
    size: SplashSize,
}

#[derive(Clone, Copy)]
pub enum SplashSize {
    XL,
    L,
    M,
    S,
    XS,
}

impl MotionBlurSplash {
    pub fn new(size: SplashSize) -> Self {
        Self { frame: 0, size }
    }

    pub fn tick(&mut self) {
        self.frame = (self.frame + 1) % 24; // 24 frames pour un cycle complet
    }

    pub fn render(&self, area: Rect, buf: &mut Buffer) {
        let content = match self.size {
            SplashSize::XS => self.render_xs(),
            SplashSize::S => self.render_s(),
            SplashSize::M => self.render_m(),
            SplashSize::L => self.render_l(),
            SplashSize::XL => self.render_xl(),
        };

        let paragraph = Paragraph::new(content)
            .alignment(Alignment::Center)
            .style(Style::default().fg(Color::Rgb(137, 180, 250))); // Blue

        paragraph.render(area, buf);
    }

    fn animated_trail(&self) -> String {

            .cycle()
            .skip(offset)
            .take(base.chars().count())
            .collect()
    }

fn detect_splash_size(terminal_width: u16) -> SplashSize {
    match terminal_width {
        0..=39 => SplashSize::XS,
        40..=59 => SplashSize::S,
        60..=79 => SplashSize::M,
        80..=119 => SplashSize::L,
