```rust
    positions: Vec<(f32, f32)>,  // (x, y) normalized 0.0-1.0

}

impl ParticleState {
    fn new(count: usize) -> Self {
        let mut rng = rand::thread_rng();
        Self {
            positions: (0..count)
                .map(|_| (rng.gen(), rng.gen()))
                .collect(),
            velocities: (0..count)
                .map(|_| (rng.gen_range(-0.02..0.02), rng.gen_range(-0.01..0.01)))
                .collect(),
            tick: 0,
        }
    }

    fn update(&mut self) {
        self.tick += 1;
        for i in 0..self.positions.len() {
            self.positions[i].0 += self.velocities[i].0;
            self.positions[i].1 += self.velocities[i].1;

            // Bounce off edges
            if self.positions[i].0 < 0.0 || self.positions[i].0 > 1.0 {
                self.velocities[i].0 *= -1.0;
            }
            if self.positions[i].1 < 0.0 || self.positions[i].1 > 1.0 {
                self.velocities[i].1 *= -1.0;
            }
        }
    }
```

```rust
use ratatui::{
    prelude::*,
    widgets::{Block, Paragraph},
};
use rand::Rng;

pub struct ParticleSplash {
    particles: Vec<Particle>,
    tick: usize,
    size: SplashSize,
}

struct Particle {
    x: f32,
    y: f32,
    vx: f32,
    vy: f32,
    intensity: ParticleIntensity,
}

#[derive(Clone, Copy)]
enum ParticleIntensity {
    Dim,    // Â·
    Medium, // âˆ™
    Bright, // â€¢
}

impl ParticleIntensity {
    fn to_char(&self) -> char {
        match self {
            Self::Dim => 'Â·',
            Self::Medium => 'âˆ™',
            Self::Bright => 'â€¢',
        }
    }

    fn to_color(&self) -> Color {
        match self {
            Self::Dim => Color::Rgb(108, 112, 134),
            Self::Medium => Color::Rgb(147, 153, 178),
            Self::Bright => Color::Rgb(205, 214, 244),
        }
    }
}

impl ParticleSplash {
    pub fn new(size: SplashSize, particle_count: usize) -> Self {
        let mut rng = rand::thread_rng();
        let particles = (0..particle_count)
            .map(|_| Particle {
                x: rng.gen(),
                y: rng.gen(),
                vx: rng.gen_range(-0.01..0.01),
                vy: rng.gen_range(-0.005..0.005),
                intensity: match rng.gen_range(0..3) {
                    0 => ParticleIntensity::Dim,
                    1 => ParticleIntensity::Medium,
                    _ => ParticleIntensity::Bright,
                },
            })
            .collect();

        Self { particles, tick: 0, size }
    }

    pub fn tick(&mut self) {
        self.tick += 1;

        for p in &mut self.particles {
            p.x += p.vx;
            p.y += p.vy;

            // Soft bounce
            if p.x < 0.0 || p.x > 1.0 { p.vx *= -0.9; }
            if p.y < 0.0 || p.y > 1.0 { p.vy *= -0.9; }

            // Clamp
            p.x = p.x.clamp(0.0, 1.0);
            p.y = p.y.clamp(0.0, 1.0);

            // Random intensity changes (twinkling)
            if self.tick % 20 == 0 {
                let mut rng = rand::thread_rng();
                if rng.gen_bool(0.3) {
                    p.intensity = match rng.gen_range(0..3) {
                        0 => ParticleIntensity::Dim,
                        1 => ParticleIntensity::Medium,
                        _ => ParticleIntensity::Bright,
                    };
                }
            }
        }
    }

    pub fn render(&self, area: Rect, buf: &mut Buffer) {
        // Render particles
        for p in &self.particles {
            let x = (p.x * area.width as f32) as u16 + area.x;
            let y = (p.y * area.height as f32) as u16 + area.y;

            if x < area.right() && y < area.bottom() {
                buf.get_mut(x, y)
                    .set_char(p.intensity.to_char())
                    .set_fg(p.intensity.to_color());
            }
        }

        // Render logo (centered)
        self.render_logo(area, buf);
    }

    fn render_border_line(&self) -> String {
        let particles = " Â·  âˆ™  â€¢ ";
        let line = "â”".repeat(30);
        format!("{}{}{}", particles, line, particles.chars().rev().collect::<String>())
    }
}
```
